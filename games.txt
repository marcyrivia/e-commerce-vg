CREATE TABLE CATEGORIE(
   CAT_ID INT auto_increment,
   CAT_DESCRIPTION VARCHAR(150),
   CAT_TYPE VARCHAR(50),
   PRIMARY KEY(CAT_ID)
);

CREATE TABLE FOURNISSEUR(
   FOUR_ID INT auto_increment,
   FOUR_MAIL VARCHAR(50),
   FOUR_ADRESSE VARCHAR(50),
   FOUR_PHONE VARCHAR(50),
   FOUR_ADRESSE_FAC VARCHAR(50),
   FOUR_NAME VARCHAR(50),
   PRIMARY KEY(FOUR_ID)
);

CREATE TABLE STUDIOS(
   S_ID INT auto_increment,
   S_NAME VARCHAR(50),
   S_NATIONALITY VARCHAR(50),
   PRIMARY KEY(S_ID)
);

CREATE TABLE CLIENT(
   CLIENT_ID INT auto_increment,
   CLIENT_ADRESSE_FACT VARCHAR(50),
   CLIENT_ADRESSE_LIVR VARCHAR(50),
   CLIENT_LASTNAME VARCHAR(50),
   CLIENT_FIRSTNAME VARCHAR(50),
   CLIENT_MAIL VARCHAR(50),
   CLIENT_ADRESSE VARCHAR(50),
   CLIENT_PHONE VARCHAR(50),
   PRIMARY KEY(CLIENT_ID)
);

CREATE TABLE COMMANDES(
   COM_ID INT auto_increment,
   COM_LIVRAISON VARCHAR(50),
   CLIENT_ID INT NOT NULL,
   PRIMARY KEY(COM_ID),
   FOREIGN KEY(CLIENT_ID) REFERENCES CLIENT(CLIENT_ID)
);

CREATE TABLE GAMES(
   G_ID INT auto_increment,
   G_TITLE VARCHAR(50) NOT NULL,
   G_PRICES VARCHAR(50),
   G_DESCRIPTION VARCHAR(255),
   G_QUANTITE INT,
   G_PLATEFORMES VARCHAR(50),
   G_IMAGE VARCHAR(150),
   FOUR_ID INT NOT NULL,
   PRIMARY KEY(G_ID),
   FOREIGN KEY(FOUR_ID) REFERENCES FOURNISSEUR(FOUR_ID)
);

CREATE TABLE AVIS(
   AVIS_ID INT auto_increment,
   AVIS_CLIENTS VARCHAR(50),
   AVIS_NOTE INT,
   G_ID INT NOT NULL,
   CLIENT_ID INT NOT NULL,
   PRIMARY KEY(AVIS_ID),
   FOREIGN KEY(G_ID) REFERENCES GAMES(G_ID),
   FOREIGN KEY(CLIENT_ID) REFERENCES CLIENT(CLIENT_ID)
);

CREATE TABLE UN_JEU_A_UN_STUDIO(
   G_ID INT,
   S_ID INT,
   PRIMARY KEY(G_ID, S_ID),
   FOREIGN KEY(G_ID) REFERENCES GAMES(G_ID),
   FOREIGN KEY(S_ID) REFERENCES STUDIOS(S_ID)
);

CREATE TABLE UN_JEU_A_UNE_CATEGORIE(
   G_ID INT,
   CAT_ID INT,
   PRIMARY KEY(G_ID, CAT_ID),
   FOREIGN KEY(G_ID) REFERENCES GAMES(G_ID),
   FOREIGN KEY(CAT_ID) REFERENCES CATEGORIE(CAT_ID)
);

CREATE TABLE UN_JEU_A_UNE_COMMANDE(
   G_ID INT,

   PRIMARY KEY(G_ID, COM_ID),
   FOREIGN KEY(G_ID) REFERENCES GAMES(G_ID),
   FOREIGN KEY(COM_ID) REFERENCES COMMANDES(COM_ID)
);
